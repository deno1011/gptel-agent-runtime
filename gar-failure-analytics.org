#+TITLE: gar-failure-analytics: visible failure-pattern aggregation
#+PROPERTY: header-args:emacs-lisp :tangle gar-failure-analytics.el :lexical t

Module of [[file:gptel-agent-runtime.org][gptel-agent-runtime]]. Closes the visibility gap on the
failure side of the learning loop.

The substrate already records every failure: verifier verdicts
(`gar-verifier'), failed action results, trajectories with
=outcome 'failure=. PR 3 (refinement) acts on consistent
failures by proposing rewrites; PR 17 brings those refinement
candidates into the unified review. But /nothing/ shows the user
the raw failure landscape -- "what fails most often", "which
tool errors most", "which reasons recur" -- until refinement
fires and writes a candidate.

For users in the early days of the trajectory archive (before
patterns are clear enough to trigger automatic refinement), this
visibility is the difference between "I trust the system is
learning" and "I have to take it on faith".

This module:

1. Aggregates failures from
   =--last-verifier-verdicts= (recent ring of 50, in-memory) and
   =--trajectories= (recent ring of up to 200, in-memory).
2. Groups them by /tool/ and by /reason pattern/ (substring
   match on common shapes: =not found=, =ambiguous=, =denied=,
   =timeout=, =void=, etc.).
3. Provides a compact summary for the mission-control dashboard
   and a detailed report buffer via
   =M-x gptel-agent-runtime-failure-report=.

* Execution flow

This section explains how the functions above participate in runtime execution. It is Org prose outside source blocks, so tangling preserves behavior.

This module summarizes failure patterns for the dashboard and reports.

- It gathers failure evidence from verifier verdicts and stored trajectories.
- Reason classification maps raw messages to categories like not-found, ambiguous, policy-denied, timeout, incomplete, or quarantined.
- Aggregation counts top reasons and tools.
- Remediation helpers map common reason/tool pairs to suggested next actions.
- Mission control uses the summary; `gptel-agent-runtime-failure-report' opens the detailed report buffer.

#+begin_src emacs-lisp
(defun gptel-agent-runtime--failure-analytics-classify-reason (text)
  "Return the bucket label for failure TEXT, or `\"other\"' when nothing matches."
  (if (or (null text) (string-empty-p text))
      "other"
    (let ((case-fold-search t)
          (result nil))
      (catch 'done
        (dolist (entry gptel-agent-runtime--failure-analytics-reason-patterns)
          (when (string-match-p (cdr entry) text)
            (setq result (car entry))
            (throw 'done nil))))
      (or result "other"))))

(defun gptel-agent-runtime--failure-analytics-failures-from-verdicts ()
  "Walk the in-memory ring of verifier verdicts and return failure
plists -- one per `:passed nil' verdict -- with shape
`(:tool TOOL :reason REASON :step STEP :at TIMESTAMP)'."
  (let ((collected '()))
    (dolist (entry (and (boundp 'gptel-agent-runtime--last-verifier-verdicts)
                        gptel-agent-runtime--last-verifier-verdicts))
      (let* ((at (car entry))
             (verdict (cdr entry)))
        (when (and verdict (not (plist-get verdict :passed)))
          (push (list :tool (or (plist-get verdict :tool) "<unknown>")
                      :reason (gptel-agent-runtime--failure-analytics-classify-reason
                               (plist-get verdict :reason))
                      :reason-text (plist-get verdict :reason)
                      :step (or (plist-get verdict :step) "")
                      :at at
                      :source 'verifier)
                collected))))
    (nreverse collected)))

(defun gptel-agent-runtime--failure-analytics-failures-from-trajectories ()
  "Walk the in-memory trajectory ring and return failure plists.
For each trajectory with `outcome 'failure', extracts the last
failing step (its tool + result-error) plus the goal."
  (let ((collected '())
        (window gptel-agent-runtime-failure-analytics-window))
    (dolist (traj (and (boundp 'gptel-agent-runtime--trajectories)
                       (let ((traj-list gptel-agent-runtime--trajectories))
                         (cl-subseq traj-list 0
                                    (min window (length traj-list))))))
      (let ((outcome (gptel-agent-runtime-trajectory-outcome traj)))
        (when (eq outcome 'failure)
          (let* ((steps (gptel-agent-runtime-trajectory-steps traj))
                 (failing-step
                  (cl-find-if
                   (lambda (s)
                     (let ((status (gptel-agent-runtime-trajectory-step-result-status
                                    s)))
                       (and status (not (eq status 'ok)))))
                   steps))
                 (tool (and failing-step
                            (gptel-agent-runtime-trajectory-step-suggested-tool
                             failing-step)))
                 (err (and failing-step
                           (gptel-agent-runtime-trajectory-step-result-error
                            failing-step))))
            (push (list :tool (or tool "<unknown>")
                        :reason (gptel-agent-runtime--failure-analytics-classify-reason
                                 err)
                        :reason-text err
                        :goal (gptel-agent-runtime-trajectory-goal traj)
                        :source 'trajectory)
                  collected)))))
    (nreverse collected)))

(defun gptel-agent-runtime--failure-analytics-all-failures ()
  "Return the union of verdict + trajectory failures."
  (append (gptel-agent-runtime--failure-analytics-failures-from-verdicts)
          (gptel-agent-runtime--failure-analytics-failures-from-trajectories)))

(defun gptel-agent-runtime--failure-analytics-top-counts (key failures)
  "Group FAILURES by KEY (`:tool' or `:reason'), return alist of
(KEY-VALUE . COUNT) sorted by descending count, capped at
`failure-analytics-top-n'."
  (let ((counts (make-hash-table :test #'equal))
        (n gptel-agent-runtime-failure-analytics-top-n))
    (dolist (f failures)
      (let ((v (or (plist-get f key) "<unknown>")))
        (puthash v (1+ (gethash v counts 0)) counts)))
    (let* ((pairs nil))
      (maphash (lambda (k v) (push (cons k v) pairs)) counts)
      (setq pairs (sort pairs (lambda (a b) (> (cdr a) (cdr b)))))
      (cl-subseq pairs 0 (min n (length pairs))))))

(defun gptel-agent-runtime-failure-analytics-stats ()
  "Return a stats plist:
`(:total N :by-tool ((TOOL . COUNT) ...) :by-reason ((REASON . COUNT) ...))'.
Suitable for both the mission-control summary and the detailed report."
  (let* ((failures (gptel-agent-runtime--failure-analytics-all-failures))
         (total (length failures))
         (by-tool (gptel-agent-runtime--failure-analytics-top-counts
                   :tool failures))
         (by-reason (gptel-agent-runtime--failure-analytics-top-counts
                     :reason failures)))
    (list :total total
          :by-tool by-tool
          :by-reason by-reason)))

#+begin_src emacs-lisp
(defun gptel-agent-runtime-failure-remediation-suggestion (reason &optional tool)
  "Look up a remediation suggestion for REASON (and optional TOOL).

Tries (REASON . TOOL) first, falls back to (REASON . nil) so a
suggestion that applies regardless of tool still matches.  Returns
the suggestion string or nil when no entry exists."
  (or (and tool
           (alist-get (cons reason tool)
                      gptel-agent-runtime-failure-remediation-suggestions
                      nil nil #'equal))
      (alist-get (cons reason nil)
                 gptel-agent-runtime-failure-remediation-suggestions
                 nil nil #'equal)))

(defun gptel-agent-runtime--failure-analytics-top-tool-for-reason (reason failures)
  "Return the most-failing TOOL name for REASON among FAILURES, or nil."
  (let ((counts (make-hash-table :test #'equal)))
    (dolist (f failures)
      (when (equal reason (plist-get f :reason))
        (let ((tool (or (plist-get f :tool) "<unknown>")))
          (puthash tool (1+ (gethash tool counts 0)) counts))))
    (let (best best-count)
      (maphash (lambda (k v)
                 (when (or (null best) (> v best-count))
                   (setq best k best-count v)))
               counts)
      best)))

(defun gptel-agent-runtime-failure-analytics-summary ()
  "Return a multi-line string for the mission-control `Failure
patterns' section.  Each top reason gets a `→ <suggestion>' line when
a remediation is registered in `failure-remediation-suggestions'."
  (let* ((stats (gptel-agent-runtime-failure-analytics-stats))
         (total (plist-get stats :total))
         (by-tool (plist-get stats :by-tool))
         (by-reason (plist-get stats :by-reason))
         (all-failures (gptel-agent-runtime--failure-analytics-all-failures))
         (format-tool-line
          (lambda (pair)
            (format "    %-28s  %d" (car pair) (cdr pair))))
         (format-reason-line
          (lambda (pair)
            (let* ((reason (car pair))
                   (count (cdr pair))
                   (top-tool
                    (gptel-agent-runtime--failure-analytics-top-tool-for-reason
                     reason all-failures))
                   (suggestion
                    (gptel-agent-runtime-failure-remediation-suggestion
                     reason top-tool)))
              (concat
               (format "    %-28s  %d" reason count)
               (when suggestion
                 (format "\n      → %s" suggestion)))))))
    (if (zerop total)
        "  (no failures recorded yet -- the substrate is healthy or empty)"
      (concat
       (format "  Total recent failures: %d\n" total)
       "  By tool:\n"
       (if by-tool
           (mapconcat format-tool-line by-tool "\n")
         "    (no per-tool data)")
       "\n  By reason:\n"
       (if by-reason
           (mapconcat format-reason-line by-reason "\n")
         "    (no per-reason data)")
       "\n  Report: M-x gptel-agent-runtime-failure-report"))))

(defcustom gptel-agent-runtime-failure-report-buffer-name
  "*gptel-agent-failure-report*"
  "Buffer name for the detailed failure report."
  :type 'string
  :group 'gptel-agent-runtime)

;;;###autoload
(defun gptel-agent-runtime-failure-report ()
  "Open a detailed failure-pattern report covering the recent
trajectory + verifier-verdict ring.  Read-only buffer."
  (interactive)
  (let* ((stats (gptel-agent-runtime-failure-analytics-stats))
         (failures (gptel-agent-runtime--failure-analytics-all-failures))
         (buf (get-buffer-create
               gptel-agent-runtime-failure-report-buffer-name)))
    (with-current-buffer buf
      (let ((inhibit-read-only t))
        (erase-buffer)
        (insert (format "Failure-pattern report  (recent %d records)\n\n"
                        (plist-get stats :total)))
        (insert "By tool:\n")
        (if (plist-get stats :by-tool)
            (dolist (pair (plist-get stats :by-tool))
              (insert (format "  %-30s  %d\n" (car pair) (cdr pair))))
          (insert "  (none)\n"))
        (insert "\nBy reason:\n")
        (if (plist-get stats :by-reason)
            (dolist (pair (plist-get stats :by-reason))
              (let* ((reason (car pair))
                     (count (cdr pair))
                     (top-tool
                      (gptel-agent-runtime--failure-analytics-top-tool-for-reason
                       reason failures))
                     (suggestion
                      (gptel-agent-runtime-failure-remediation-suggestion
                       reason top-tool)))
                (insert (format "  %-30s  %d\n" reason count))
                (when suggestion
                  (insert (format "    Suggested: %s\n" suggestion)))))
          (insert "  (none)\n"))
        (insert "\nRecent failure messages (last 20):\n\n")
        (dolist (f (cl-subseq failures 0 (min 20 (length failures))))
          (insert (format "  [%s] tool=%s reason=%s\n"
                          (or (plist-get f :source) "?")
                          (or (plist-get f :tool) "?")
                          (or (plist-get f :reason) "?")))
          (when (plist-get f :reason-text)
            (insert (format "      %s\n"
                            (string-trim
                             (substring
                              (or (plist-get f :reason-text) "")
                              0 (min 200
                                     (length (or (plist-get f :reason-text)
                                                 "")))))))))
        (goto-char (point-min))
        (special-mode)))
    (display-buffer buf)))
