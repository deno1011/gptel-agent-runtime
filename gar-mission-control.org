#+TITLE: gar-mission-control: unified runtime dashboard
#+PROPERTY: header-args:emacs-lisp :tangle gar-mission-control.el :lexical t

Module of [[file:gptel-agent-runtime.org][gptel-agent-runtime]]. The runtime's read-only observation
surface: a single buffer that summarizes the live state of every
submodule -- substrate (tick, event pump, subscribers, capability
enforcement), policy (active preset, confirmation/wrap flags), recent
events, recent evidence with taint, quarantine size + promoted IDs,
canary pass/fail summary, skeptic verdicts, exploration & learning
(novelty / synthesis / playbooks), and per-agent capability
allowlists. Subscribes to =tick= on first render so subsequent ticks
refresh the buffer in place.

This is the final piece of the gar-safety sub-split (2026-05-27 through
2026-05-28). After this PR, =gar-safety= is removed entirely; the
original responsibilities now live across five focused modules:
=gar-canaries=, =gar-quarantine=, =gar-skeptic=, =gar-policy=, and
this one.

See also:
- [[file:gptel-agent-runtime.org][gptel-agent-runtime]] (master entrypoint that requires this module)
- [[file:gar-substrate.org][gar-substrate]] (tick / event pump / evidence trace consulted here)
- [[file:gar-policy.org][gar-policy]] (policy preset + capability enforcement defcustoms read here)
- [[file:gar-quarantine.org][gar-quarantine]] (quarantine defcustoms + accessors read here)
- [[file:gar-canaries.org][gar-canaries]] (=--last-canary-results= summarised here)
- [[file:gar-skeptic.org][gar-skeptic]] (skeptic defcustoms + recent verdicts summarised here)
- [[file:gar-memory.org][gar-memory]] (novelty / synthesis / playbook stats surfaced here)
- [[file:gar-agents.org][gar-agents]] (per-agent capability allowlists rendered here)

(defcustom gptel-agent-runtime-mission-control-buffer-name "*gptel-agent-mission-control*"
  "Buffer name used for the unified mission-control dashboard."
  :type 'string
  :group 'gptel-agent-runtime)

(defvar gptel-agent-runtime--mission-control-subscribed nil
  "Non-nil when the mission-control auto-refresh subscriber is installed.")

(defun gptel-agent-runtime--mission-control-section (title body)
  "Insert a TITLE section with BODY (a string) into the current buffer."
  (insert (format "=== %s ===\n%s\n\n" title body)))

(defun gptel-agent-runtime--mission-control-recent-events (limit)
  "Return a string with the LIMIT most recent dispatched events for the dashboard."
  (let* ((entries gptel-agent-runtime--last-dispatched-events)
         (n (min (or limit 8) (length entries))))
    (if (zerop n)
        "  (no events dispatched yet)"
      (mapconcat
       (lambda (entry)
         (let ((evt (plist-get entry :event)))
           (format "  %s  %s  handlers=%d errors=%d"
                   (gptel-agent-runtime-event-created-at evt)
                   (gptel-agent-runtime-event-type evt)
                   (length (plist-get entry :handlers))
                   (length (plist-get entry :errors)))))
       (cl-subseq entries 0 n)
       "\n"))))

(defun gptel-agent-runtime--mission-control-recent-evidence (limit)
  "Return a string with the LIMIT most recent evidence records."
  (let* ((trace gptel-agent-runtime--evidence-trace)
         (n (min (or limit 6) (length trace))))
    (if (zerop n)
        "  (no evidence yet)"
      (mapconcat
       (lambda (ev)
         (format "  %s [%s/%s] %s"
                 (gptel-agent-runtime-evidence-id ev)
                 (gptel-agent-runtime-evidence-source-type ev)
                 (gptel-agent-runtime-evidence-taint ev)
                 (gptel-agent-runtime--shorten
                  (gptel-agent-runtime-evidence-text ev) 100)))
       (cl-subseq trace 0 n)
       "\n"))))

(defun gptel-agent-runtime--mission-control-canary-summary ()
  "Return a string describing the most recent canary run."
  (if (null gptel-agent-runtime--last-canary-results)
      "  (canaries have not been run; M-x gptel-agent-runtime-run-injection-canaries)"
    (let ((pass (cl-count-if (lambda (r) (nth 1 r))
                             gptel-agent-runtime--last-canary-results))
          (total (length gptel-agent-runtime--last-canary-results))
          (fails (cl-remove-if (lambda (r) (nth 1 r))
                               gptel-agent-runtime--last-canary-results)))
      (concat (format "  %d/%d passed" pass total)
              (when fails
                (concat "\n  failing: "
                        (mapconcat (lambda (r) (nth 0 r)) fails ", ")))))))

(defun gptel-agent-runtime-mission-control ()
  "Open the unified mission-control dashboard buffer.
Shows the OpenClaw tick, idle-pump state, recent dispatched events, active
policy preset, recent evidence flow with taint, quarantine size, canary
status, and the registered agent capability allowlist."
  (interactive)
  (with-current-buffer (get-buffer-create
                        gptel-agent-runtime-mission-control-buffer-name)
    (let ((inhibit-read-only t))
    (erase-buffer)
    (insert (format "gptel-agent-runtime mission control\nRendered at: %s\n\n"
                    (gptel-agent-runtime--timestamp)))
    (gptel-agent-runtime--mission-control-section
     "Substrate"
     (format "  Tick: %s\n  Idle pump: %s (every %ds)\n  Schema: %s\n  Subscribers: %d types\n  Capability enforcement: %s"
             (or gptel-agent-runtime-tick-counter 0)
             (if gptel-agent-runtime--idle-pump-timer "ON" "off")
             gptel-agent-runtime-idle-pump-interval
             gptel-agent-runtime-state-schema-version
             (length gptel-agent-runtime--event-subscribers)
             (if gptel-agent-runtime-capability-enforcement-enabled "ON" "off")))
    (gptel-agent-runtime--mission-control-section
     "Policy"
     (format "  Preset: %s\n  Confirm for risky: %s\n  Auto-execute safe: %s\n  Wrap untrusted: %s\n  Quarantine untrusted: %s (pre-flight=%s)"
             gptel-agent-runtime-policy-preset
             gptel-agent-runtime-require-confirmation-for-risky-actions
             gptel-agent-runtime-auto-execute-safe-actions
             gptel-agent-runtime-wrap-untrusted-context
             gptel-agent-runtime-quarantine-untrusted-output
             gptel-agent-runtime-quarantine-pre-flight-enabled))
    (gptel-agent-runtime--mission-control-section
     "Recent events"
     (gptel-agent-runtime--mission-control-recent-events 8))
    (gptel-agent-runtime--mission-control-section
     "Recent evidence"
     (gptel-agent-runtime--mission-control-recent-evidence 6))
    (gptel-agent-runtime--mission-control-section
     "Quarantine"
     (let* ((q (gptel-agent-runtime-quarantined-evidence)))
       (if (null q)
           "  (no quarantined evidence)"
         (concat (format "  %d items quarantined; %d promoted IDs\n"
                         (length q)
                         (length gptel-agent-runtime--promoted-evidence-ids))
                 (mapconcat (lambda (e)
                              (format "  - %s [%s]"
                                      (gptel-agent-runtime-evidence-id e)
                                      (gptel-agent-runtime-evidence-source-type e)))
                            (cl-subseq q 0 (min 5 (length q)))
                            "\n")))))
    (gptel-agent-runtime--mission-control-section
     "Injection canaries"
     (gptel-agent-runtime--mission-control-canary-summary))
    (gptel-agent-runtime--mission-control-section
     "Skeptic"
     (format "  Enabled: %s   Mode: %s   Budget: %dms\n  Trigger risks: %s\n  Trigger caps: %s\n  Recent verdicts: %d\n%s"
             gptel-agent-runtime-skeptic-enabled
             gptel-agent-runtime-skeptic-mode
             gptel-agent-runtime-skeptic-budget-ms
             gptel-agent-runtime-skeptic-trigger-risks
             gptel-agent-runtime-skeptic-trigger-caps
             (length gptel-agent-runtime--last-skeptic-verdicts)
             (if (null gptel-agent-runtime--last-skeptic-verdicts)
                 "  (no verdicts yet)"
               (mapconcat
                (lambda (entry)
                  (let ((v (cdr entry)))
                    (format "  %s  tool=%s  risk=%s  concerns=%d"
                            (car entry)
                            (plist-get v :tool)
                            (plist-get v :risk)
                            (length (plist-get v :concerns)))))
                (cl-subseq gptel-agent-runtime--last-skeptic-verdicts
                           0 (min 5 (length gptel-agent-runtime--last-skeptic-verdicts)))
                "\n"))))
    (gptel-agent-runtime--mission-control-section
     "Verifier"
     (format "  Mode: %s   Budget: %dms   Max retries: %d\n  Trigger risks: %s\n  Recent verdicts: %d\n%s"
             gptel-agent-runtime-verifier-mode
             gptel-agent-runtime-verifier-budget-ms
             gptel-agent-runtime-verifier-max-retries
             gptel-agent-runtime-verifier-trigger-risks
             (length gptel-agent-runtime--last-verifier-verdicts)
             (if (null gptel-agent-runtime--last-verifier-verdicts)
                 "  (no verdicts yet)"
               (mapconcat
                (lambda (entry)
                  (let ((v (cdr entry)))
                    (format "  %s  tool=%s  passed=%s  conf=%.2f  %s"
                            (car entry)
                            (plist-get v :tool)
                            (plist-get v :passed)
                            (or (plist-get v :confidence) 0.0)
                            (gptel-agent-runtime--shorten
                             (or (plist-get v :reason) "") 60))))
                (cl-subseq gptel-agent-runtime--last-verifier-verdicts
                           0 (min 5
                                  (length gptel-agent-runtime--last-verifier-verdicts)))
                "\n"))))
    (gptel-agent-runtime--mission-control-section
     "Exploration & learning"
     (format "  Novelty threshold: %.2f   Min tokens: %d\n  Strategy synthesis: %s   Interval: every %d ticks\n  Candidate playbooks pending: %d\n  Top playbooks (by success rate):\n%s"
             gptel-agent-runtime-novelty-threshold
             gptel-agent-runtime-novelty-min-tokens
             gptel-agent-runtime-strategy-synthesis-enabled
             gptel-agent-runtime-strategy-synthesis-interval-ticks
             (length (or (gptel-agent-runtime-list-playbook-candidates) '()))
             (let ((top (gptel-agent-runtime-rank-playbooks-by-success 5)))
               (if (null top)
                   "    (no playbooks registered)"
                 (mapconcat
                  (lambda (pb)
                    (let ((rate (gptel-agent-runtime-playbook-success-rate pb)))
                      (format "    %s  %s"
                              (or (gptel-agent-runtime-playbook-id pb)
                                  (gptel-agent-runtime-playbook-summary pb))
                              (if rate (format "%.0f%%" (* 100 rate)) "unused"))))
                  top "\n")))))
    (gptel-agent-runtime--mission-control-section
     "Trajectories"
     (let* ((traj gptel-agent-runtime--trajectories)
            (n (length traj))
            (outcomes (let (acc)
                        (dolist (t (cl-subseq traj 0 (min n 20)))
                          (let ((o (gptel-agent-runtime-trajectory-outcome t)))
                            (setf (alist-get o acc 0) (1+ (alist-get o acc 0)))))
                        acc)))
       (if (zerop n)
           "  (no trajectories recorded yet)"
         (format "  In-memory: %d   Recent outcomes: %s\n  Recent goals:\n%s"
                 n
                 (or outcomes "—")
                 (mapconcat
                  (lambda (t)
                    (format "    [%s] %s -- %s"
                            (or (gptel-agent-runtime-trajectory-outcome t) "?")
                            (or (gptel-agent-runtime-trajectory-finalized-at t) "?")
                            (gptel-agent-runtime--shorten
                             (or (gptel-agent-runtime-trajectory-goal t) "") 60)))
                  (cl-subseq traj 0 (min n 5))
                  "\n")))))
    (gptel-agent-runtime--mission-control-section
     "Playbook refinement"
     (format "  Mode: %s   Window: %d   Threshold: %.0f%%\n  Recent refinements: %d\n%s"
             gptel-agent-runtime-refine-mode
             gptel-agent-runtime-refine-window
             (* 100 gptel-agent-runtime-refine-failure-threshold)
             (length gptel-agent-runtime--last-refinements)
             (if (null gptel-agent-runtime--last-refinements)
                 "  (no refinements yet)"
               (mapconcat
                (lambda (r)
                  (format "    %s  playbook=%s  candidate=%s  rate=%.0f%%"
                          (or (plist-get r :at) "?")
                          (or (plist-get r :playbook) "?")
                          (or (plist-get r :candidate) "?")
                          (* 100 (or (plist-get r :failure-rate) 0.0))))
                (cl-subseq gptel-agent-runtime--last-refinements
                           0 (min 5 (length gptel-agent-runtime--last-refinements)))
                "\n"))))
    (gptel-agent-runtime--mission-control-section
     "Memory index (SQLite)"
     (let ((stats (and (fboundp 'gptel-agent-runtime-sqlite-stats)
                       (gptel-agent-runtime-sqlite-stats))))
       (if (null stats)
           "  (SQLite index unavailable; flat-file trajectories only)"
         (format "  Trajectories: %d   Embeddings: %d   FTS5: %s\n  File: %s"
                 (or (plist-get stats :trajectories) 0)
                 (or (plist-get stats :embeddings) 0)
                 (if (plist-get stats :fts5-available) "ON" "off (LIKE fallback)")
                 (gptel-agent-runtime--shorten
                  (or (plist-get stats :file) "") 60)))))
    (gptel-agent-runtime--mission-control-section
     "Experiments (A/B)"
     (let* ((exps (and (boundp 'gptel-agent-runtime--experiments)
                       gptel-agent-runtime--experiments))
            (running (cl-count-if (lambda (e)
                                    (eq (gptel-agent-runtime-experiment-status e)
                                        'running))
                                  (or exps '()))))
       (if (null exps)
           "  (no experiments)"
         (format "  Total: %d   Running: %d\n%s"
                 (length exps) running
                 (mapconcat
                  (lambda (e)
                    (format "    [%s] playbook=%s  orig=%d/%d  cand=%d/%d"
                            (gptel-agent-runtime-experiment-status e)
                            (gptel-agent-runtime-experiment-playbook-id e)
                            (or (gptel-agent-runtime-experiment-original-successes
                                 e) 0)
                            (+ (or (gptel-agent-runtime-experiment-original-successes
                                    e) 0)
                               (or (gptel-agent-runtime-experiment-original-failures
                                    e) 0))
                            (or (gptel-agent-runtime-experiment-candidate-successes
                                 e) 0)
                            (+ (or (gptel-agent-runtime-experiment-candidate-successes
                                    e) 0)
                               (or (gptel-agent-runtime-experiment-candidate-failures
                                    e) 0))))
                  (cl-subseq exps 0 (min 5 (length exps)))
                  "\n")))))
    (gptel-agent-runtime--mission-control-section
     "Agents / capability allowlists"
     (if (null gptel-agent-runtime-agent-registry)
         "  (no agents registered)"
       (mapconcat
        (lambda (a)
          (format "  %s [%s]  caps=%s"
                  (gptel-agent-runtime-agent-name a)
                  (gptel-agent-runtime-agent-role a)
                  (or (gptel-agent-runtime-agent-allowed-caps a) '(any))))
        gptel-agent-runtime-agent-registry
        "\n")))
    (goto-char (point-min))
    (special-mode)))
  (unless gptel-agent-runtime--mission-control-subscribed
    (gptel-agent-runtime-subscribe
     'tick
     (lambda (_e)
       (when (get-buffer gptel-agent-runtime-mission-control-buffer-name)
         ;; Refresh in place without stealing window focus.
         (save-window-excursion
           (gptel-agent-runtime-mission-control)))))
    (setq gptel-agent-runtime--mission-control-subscribed t))
  (display-buffer gptel-agent-runtime-mission-control-buffer-name))

#+begin_src emacs-lisp
;; Inline editor for `gptel-agent-runtime-tool-policy'. The policy alist
;; lives in gar-core; the active preset and the per-tool capability manifest
;; live in gar-policy. This buffer renders one row per known tool with the
;; effective :default / :confirm / :taint and which layer the value comes
;; from (user override / preset / default / none).

(defvar gptel-agent-runtime-tool-capabilities)
(defvar gptel-agent-runtime-default-tool-policy)
(defvar gptel-agent-runtime-tool-policy)

(declare-function gptel-agent-runtime--policy-for-tool
                  "gar-policy" (tool-name))

(defcustom gptel-agent-runtime-tool-policy-editor-buffer-name
  "*gptel-agent-tool-policy*"
  "Buffer name for the tool policy editor."
  :type 'string
  :group 'gptel-agent-runtime)

(defconst gptel-agent-runtime--tool-policy-confirm-cycle
  '(nil write shell destructive always)
  "Cycle order for the :confirm key in the tool policy editor.")

(defconst gptel-agent-runtime--tool-policy-default-cycle
  '(allow deny)
  "Cycle order for the :default key in the tool policy editor.")

(defconst gptel-agent-runtime--tool-policy-taint-cycle
  '(nil trusted untrusted)
  "Cycle order for the :taint key in the tool policy editor.
A nil entry means \"do not override\" — fall back to the default-policy or
preset entry.")

(defun gptel-agent-runtime--tool-policy-known-tools ()
  "Return the union of tool names from the capability manifest and policy alists."
  (let* ((from-caps (mapcar #'car gptel-agent-runtime-tool-capabilities))
         (from-default (mapcar (lambda (e)
                                 (format "%s" (car e)))
                               gptel-agent-runtime-default-tool-policy))
         (from-user (mapcar (lambda (e)
                              (format "%s" (car e)))
                            gptel-agent-runtime-tool-policy)))
    (sort (cl-remove-duplicates
           (append from-caps from-default from-user)
           :test #'equal)
          #'string<)))

(defun gptel-agent-runtime--tool-policy-source (tool)
  "Return symbol describing which layer supplies TOOL's effective policy.
One of `user', `preset', `default', or `none'."
  (cond
   ((assoc tool gptel-agent-runtime-tool-policy) 'user)
   ((let* ((entry (gptel-agent-runtime--policy-for-tool tool)))
      (and entry
           (not (assoc tool gptel-agent-runtime-default-tool-policy))))
    'preset)
   ((assoc tool gptel-agent-runtime-default-tool-policy) 'default)
   (t 'none)))

(defun gptel-agent-runtime--tool-policy-row (tool)
  "Return the tabulated-list row vector for TOOL."
  (let* ((eff (gptel-agent-runtime--policy-for-tool tool))
         (default (plist-get eff :default))
         (confirm (plist-get eff :confirm))
         (taint (plist-get eff :taint))
         (caps (cdr (assoc tool gptel-agent-runtime-tool-capabilities)))
         (source (gptel-agent-runtime--tool-policy-source tool)))
    (list tool
          (vector tool
                  (format "%s" (or default 'allow))
                  (format "%s" (or confirm "-"))
                  (format "%s" (or taint "-"))
                  (format "%d" (length caps))
                  (symbol-name source)))))

(defun gptel-agent-runtime--tool-policy-rows ()
  "Return all rows for the editor, one per known tool."
  (mapcar #'gptel-agent-runtime--tool-policy-row
          (gptel-agent-runtime--tool-policy-known-tools)))

(defun gptel-agent-runtime--tool-policy-current-tool ()
  "Return the tool name on the current line, or nil."
  (tabulated-list-get-id))

(defun gptel-agent-runtime--tool-policy-current-entry (tool)
  "Return the current USER policy plist for TOOL (copy), or nil."
  (copy-sequence (cdr (assoc tool gptel-agent-runtime-tool-policy))))

(defun gptel-agent-runtime--tool-policy-set-entry (tool plist)
  "Replace TOOL's entry in `gptel-agent-runtime-tool-policy' with PLIST.
When PLIST is nil, remove the entry instead. Mutates the user override only;
the default-policy alist is untouched."
  (setq gptel-agent-runtime-tool-policy
        (cl-remove tool gptel-agent-runtime-tool-policy
                   :key #'car :test #'equal))
  (when plist
    (push (cons tool plist) gptel-agent-runtime-tool-policy)))

(defun gptel-agent-runtime--tool-policy-cycle-key (key cycle)
  "Cycle TOOL's :KEY through CYCLE in the user policy alist.
Called from a tabulated-list-mode buffer."
  (let* ((tool (gptel-agent-runtime--tool-policy-current-tool)))
    (unless tool (user-error "No tool on this line"))
    (let* ((plist (gptel-agent-runtime--tool-policy-current-entry tool))
           (current (plist-get plist key))
           (idx (or (cl-position current cycle :test #'eq) -1))
           (next (nth (mod (1+ idx) (length cycle)) cycle))
           (new-plist (if next
                          (plist-put plist key next)
                        (gptel-agent-runtime--tool-policy-strip-key plist key))))
      (gptel-agent-runtime--tool-policy-set-entry tool new-plist)
      (tabulated-list-revert)
      (message "tool-policy: %s :%s = %s" tool key (or next "removed")))))

(defun gptel-agent-runtime--tool-policy-strip-key (plist key)
  "Return PLIST with KEY removed; return nil if the resulting plist is empty."
  (let (out)
    (while plist
      (unless (eq (car plist) key)
        (setq out (append out (list (car plist) (cadr plist)))))
      (setq plist (cddr plist)))
    (and out out)))

(defun gptel-agent-runtime-tool-policy-reset-line ()
  "Remove the user override for the tool on the current line."
  (interactive)
  (let ((tool (gptel-agent-runtime--tool-policy-current-tool)))
    (unless tool (user-error "No tool on this line"))
    (gptel-agent-runtime--tool-policy-set-entry tool nil)
    (tabulated-list-revert)
    (message "tool-policy: %s reset (user override removed)" tool)))

(defun gptel-agent-runtime-tool-policy-save ()

(defvar gptel-agent-runtime-tool-policy-editor-mode-map
  (let ((map (make-sparse-keymap)))
    (define-key map (kbd "c") #'gptel-agent-runtime-tool-policy-cycle-confirm)
    (define-key map (kbd "d") #'gptel-agent-runtime-tool-policy-cycle-default)
    (define-key map (kbd "t") #'gptel-agent-runtime-tool-policy-cycle-taint)
    (define-key map (kbd "r") #'gptel-agent-runtime-tool-policy-reset-line)
    (define-key map (kbd "s") #'gptel-agent-runtime-tool-policy-save)
    map)
  "Keymap for `gptel-agent-runtime-tool-policy-editor-mode'.")

(define-derived-mode gptel-agent-runtime-tool-policy-editor-mode
  tabulated-list-mode "GAR-Tool-Policy"
  "Major mode for editing `gptel-agent-runtime-tool-policy'.

Key bindings (inherited from `tabulated-list-mode' plus this mode's):
  c   Cycle :confirm (nil → write → shell → destructive → always → nil)
  d   Cycle :default (allow ↔ deny)
  t   Cycle :taint (nil → trusted → untrusted → nil)
  r   Reset line (remove the user override entirely)
  s   Save the policy via Customize
  g   Refresh
  q   Quit (from tabulated-list-mode)

Edits mutate `gptel-agent-runtime-tool-policy' in memory; press `s' to
persist via Customize. The Source column reports which layer supplies the
effective policy: `user' (this editor's overrides), `preset' (active
named policy preset), `default' (built-in default-tool-policy), or `none'."
  (setq tabulated-list-format
        [("Tool" 28 t)
         ("Default" 8 t)
         ("Confirm" 12 t)
         ("Taint" 12 t)
         ("Caps" 6 t)
         ("Source" 10 t)])
  (setq tabulated-list-padding 1)
  (setq tabulated-list-entries
        #'gptel-agent-runtime--tool-policy-rows)
  (tabulated-list-init-header))
